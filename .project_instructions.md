### Setup Admin User
```bash
# Connect to database and create admin role for a user
psql "postgresql://..." << EOF
DO \$\$
DECLARE
  user_id TEXT;
BEGIN
  SELECT id INTO user_id FROM "User" WHERE email = 'admin@example.com' LIMIT 1;
  INSERT INTO "AdminRole" (id, "userId", role, permissions, "createdAt", "updatedAt")
  VALUES (
    'admin-' || gen_random_uuid()::text,
    user_id,
    'admin',
    ARRAY['view_analytics', 'manage_users', 'export_data'],
    NOW(),
    NOW()
  ) ON CONFLICT ("userId") DO NOTHING;
END
\$\$;
EOF
```
